142 bool CutPool::select(const std::vector<double>& x, CutList& cuts)
143 {
144  cuts.clear();
145  CutList reducedPool;
146  // first loop: collect violated cuts and compute initial efficacy/ortho/score
147  for (CutPtr c: pool)
148  {
149  c->efficacy = c->violation(&x[0]) / c->norm; //< can be negative for slack constraints!
150  if ((!c->inUse) && (c->efficacy > minEfficacy))
151  {
152  c->ortho = 1.0;
153  reducedPool.push_back(c);
154  c->age = 0;
155  }
156  else c->age++;
157  }
158  if (isPositive(minOrtho))
159  {
160  dominiqs::numarray<double> ar(x.size());
161  // second loop: add cut with greatest score and recompoute ortho/score
162  while ( (cuts.size() < maxSeparated) && reducedPool.size() )
163  {
164  // select best cut and remove it from reduced pool
165  CutList::iterator bestItr = min_element(reducedPool.begin(), reducedPool.end(), CutCmpByScore());
166  CutPtr best = *bestItr;
167  cuts.push_back(best);
168  *bestItr = reducedPool.back();
169  reducedPool.pop_back();
170  // update ortho and recompute score
171  ar.zero();
172  best->row.scatter(&ar[0], ar.size());
173  for (CutPtr c: reducedPool)
174  {
175  double newOrtho = 1.0 - dotProduct(c->row, &ar[0]) / (best->norm * c->norm);
176  c->ortho = std::min(newOrtho, c->ortho);
177  }
178  CutList::iterator newEnd = remove_if(reducedPool.begin(), reducedPool.end(), CutOrthoFilter(minOrtho));
179  reducedPool.erase(newEnd, reducedPool.end());
180  }
181  }
182  else
183  {
184  if (reducedPool.size() > maxSeparated)
185  {
186  sort(reducedPool.begin(), reducedPool.end(), CutCmpByScore());
187  reducedPool.resize(maxSeparated);
188  }
189  cuts.swap(reducedPool);
190  }
191  return cuts.size();
192 }

194 int CutPool::purge(const std::vector<double>& x)
195 {
196  if ((memoryUsed() < maxMemory) && (size() < maxSize)) return 0;
197  CutList::size_type oldSize = pool.size();
198  std::vector<CutPtr> tmpPool;
199  tmpPool.reserve(oldSize);
200  // sort by increasing cut badness
201  for (CutPtr c: pool)
202  {
203  c->efficacy = c->violation(&x[0]) / c->norm;
204  tmpPool.push_back(c);
205  }
206  sort(tmpPool.begin(), tmpPool.end(), CutCmpForPurge());
207  // remove last half of the pool
208  for (CutList::size_type i = oldSize / 2; i < oldSize; i++)
209  {
210  pool.erase(tmpPool[i]);
211  mem -= (double)(tmpPool[i]->memoryUsed());
212  }
213  CutList::size_type newSize = pool.size();
214  return (oldSize - newSize);
215 }
