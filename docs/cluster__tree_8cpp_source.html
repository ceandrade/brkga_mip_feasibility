103 ClusterTree::ClusterTree(std::vector<CClusteringLibNode> &_plain_tree,
104  const std::vector<std::string> &object_names):
105  root(nullptr),
106  original_leaves(),
107  current_leaves(),
108  plain_tree(),
109  minimum_distance(numeric_limits<double>::max()),
110  maximum_distance(numeric_limits<double>::lowest()),
111  average_distance(0.0),
112  median_distance(0.0),
113  height(0)
114 {
115  #ifdef DEBUG
116  cout << "\n--------------------------------\n"
117  << "> Building a cluster tree..."
118  << endl;
119  #endif
120 
121  // First, take the ownership of the data.
122  plain_tree.swap(_plain_tree);
123 
124  original_leaves.reserve(object_names.size());
125  current_leaves.reserve(object_names.size());
126 
127  for(int i = 0; i < (int)object_names.size(); ++i) {
128  shared_ptr<ClusterNode> node(new ClusterNode());
129  node->original_id = i;
130  node->object_names.push_back(object_names[i]);
131  original_leaves.push_back(node);
132  current_leaves.push_back(node);
133  }
134 
135  vector<shared_ptr<ClusterNode>> intermediate_nodes(plain_tree.size(), nullptr);
136 
137  // Inner nodes have IDs using negative number. So, we need do some
138  // offsetting to get the right position in intermediate_nodes.
139  // Please, note the minus signals along this block.
140  int id_count = -1;
141  for(auto &plain_node : plain_tree) {
142  #ifdef FULLDEBUG
143  cout << "\n Creating node " << id_count << " >"
144  << " left: " << plain_node.left
145  << " right: " << plain_node.right
146  << " distance: " << plain_node.distance;
147  cout.flush();
148  #endif
149 
150  shared_ptr<ClusterNode> node(new ClusterNode());
151  node->original_id = id_count;
152  intermediate_nodes[-id_count - 1] = node;
153 
154  if(plain_node.left > -1)
155  node->left = original_leaves[plain_node.left];
156  else
157  node->left = intermediate_nodes[-plain_node.left - 1];
158 
159  if(plain_node.right > -1)
160  node->right = original_leaves[plain_node.right];
161  else
162  node->right = intermediate_nodes[-plain_node.right - 1];
163 
164  node->left->parent = node;
165  node->right->parent = node;
166 
167  node->distance = plain_node.distance;
168 
169  --id_count;
170  }
171 
172  // Usually, the last node of intermediate_nodes is the root node. But,
173  // to make sure, let's traverse the tree bottom-up from a leaf to the root.
174  weak_ptr<ClusterNode> lookup_node(original_leaves.front());
175  while(!lookup_node.lock()->parent.expired())
176  lookup_node = lookup_node.lock()->parent;
177 
178  root = lookup_node.lock();
179 
180  // Now, we adjust the heights and compute some statistics.
181  root->height = 0;
182  average_distance = root->distance;
183 
184  vector<double> distances;
185  distances.reserve(plain_tree.size());
186 
187  queue<shared_ptr<ClusterNode>> queue;
188  queue.push(root);
189  while(!queue.empty()) {
190  auto node = queue.front();
191  queue.pop();
192 
193  height = max(height, node->height);
194 
195  if(node->left != nullptr || node->right != nullptr) {
196  minimum_distance = std::min(minimum_distance, node->distance);
197  maximum_distance = std::max(maximum_distance, node->distance);
198  average_distance += node->distance;
199  distances.push_back(node->distance);
200  }
201 
202  if(node->left != nullptr) {
203  node->left->height = node->height + 1;
204  queue.push(node->left);
205  }
206 
207  if(node->right != nullptr) {
208  node->right->height = node->height + 1;
209  queue.push(node->right);
210  }
211  }
212 
213  // height = last_level + 1
214  ++height;
215 
216  average_distance /= distances.size() + 1;
217 
218  nth_element(begin(distances), begin(distances) + (distances.size() / 2), end(distances));
219  median_distance = *(begin(distances) + (distances.size() / 2));
220 }

226 void ClusterTree::compactTree(const Compaction type, const double value) {
227  queue<shared_ptr<ClusterNode>> queue;
228  queue.push(root);
229  current_leaves.clear();
230 
231  while(!queue.empty()) {
232  auto node = queue.front();
233  queue.pop();
234 
235  if((type == Compaction::BY_LEVEL && node->height < (decltype(node->height))value) ||
236  (type == Compaction::BY_DISTANCE && node->distance > value)) {
237 
238  if(node->left != nullptr)
239  queue.push(node->left);
240  if(node->right != nullptr)
241  queue.push(node->right);
242  }
243  else
244  current_leaves.push_back(node);
245  }
246 }
