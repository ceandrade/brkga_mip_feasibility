311 std::shared_ptr<ClusterTree> Clusterator::hierarchicalClustering(
312  const ClusteringObject type, const Metric metric) {
313  #ifdef DEBUG
314  cout << "\n--------------------------------\n"
315  << "> Performing hierarchical clustering..."
316  << endl;
317  #endif
318 
319  // Transpose the incidence matrix in case of constraint clustering.
320  if(type == ClusteringObject::CONSTRAINT) {
321  if(metric == Metric::L1 || metric == Metric::L2)
322  incidence_matrix.transposeInPlace();
323  else
324  weighted_incidence_matrix.transposeInPlace();
325  }
326 
327  #ifdef DEBUG
328  switch(metric) {
329  case Clusterator::Metric::L1:
330  cout << "> Computing L1...";
331  break;
332  case Clusterator::Metric::L2:
333  cout << "> Computing L2...";
334  break;
335  case Clusterator::Metric::WEIGHTED_L1:
336  cout << "> Computing weighted L1...";
337  break;
338  case Clusterator::Metric::WEIGHTED_L2:
339  cout << "> Computing weighted L2...";
340  break;
341  default:
342  break;
343  }
344  cout << endl;
345  #endif
346 
347  switch(metric) {
348  case Clusterator::Metric::L1:
349  computeL1Distance(incidence_matrix);
350  break;
351 
352  case Clusterator::Metric::L2:
353  computeL2Distance(incidence_matrix);
354  break;
355 
356  case Clusterator::Metric::WEIGHTED_L1:
357  computeL1Distance(weighted_incidence_matrix);
358  break;
359 
360  case Clusterator::Metric::WEIGHTED_L2:
361  computeL2Distance(weighted_incidence_matrix);
362  break;
363 
364  default:
365  break;
366  }
367 
368  #ifdef DEBUG
369  cout << "> Clustering..." << endl;
370  #endif
371 
372  IncidenceMatrix *inc_matrix;
373  DistanceMatrix *dist_matrix;
374 
375  if(metric == Metric::L1 || metric == Metric::L2)
376  inc_matrix = &incidence_matrix;
377  else
378  inc_matrix = &weighted_incidence_matrix;
379 
380  if(metric == Metric::SHARED) {
381  if(type == ClusteringObject::VARIABLE)
382  dist_matrix = &variables_distance;
383  else
384  dist_matrix = &constraints_distance;
385  }
386  else
387  dist_matrix = &metric_distance;
388 
389  Node* plain_tree = treecluster((int)inc_matrix->rows(), (int)inc_matrix->cols(),
390  NULL, NULL, NULL, 0, 'e', 's', dist_matrix->rawData());
391 
392  if(plain_tree == NULL)
393  throw std::runtime_error("Fail in clustering process.");
394 
395  const int TREESIZE = inc_matrix->rows() - 1;
396 
397  vector<ClusterTree::CClusteringLibNode> formatted_data;
398  formatted_data.reserve(TREESIZE);
399 
400  for(int i = 0; i < TREESIZE; ++i) {
401  auto &node = plain_tree[i];
402  formatted_data.emplace_back(node.left, node.right, node.distance);
403  }
404 
405  delete[] plain_tree;
406 
407  std::shared_ptr<ClusterTree> tree(
408  new ClusterTree(formatted_data,
409  (type == ClusteringObject::VARIABLE? variable_names :
410  constraint_names)));
411  #ifdef FULLDEBUG
412  cout << "\n\n>> Tree:\n" << *tree << endl;
413  #endif
414 
415  // Come back to the regular matrix.
416  if(type == ClusteringObject::CONSTRAINT) {
417  if(metric == Metric::L1 || metric == Metric::L2)
418  incidence_matrix.transposeInPlace();
419  else
420  weighted_incidence_matrix.transposeInPlace();
421  }
422 
423  #ifdef DEBUG
424  cout << "--------------------------------\n" << endl;
425  #endif
426 
427  return tree;
428 }

432 void Clusterator::computeL1Distance(const IncidenceMatrix &matrix) {
433  typedef decltype(matrix.rows()) IndexType;
434  const auto N = matrix.rows();
435  metric_distance.resize(N);
436 
437  // This is not the best but helps a little. We must find an algebraic
438  // form as in L2 to used the Eigen parallel machinery.
439  for(IndexType i = 0; i < N - 1; ++i) {
440  const auto &row = matrix.row(i);
441 
442  #ifdef _OPENMP
443  #pragma omp parallel for shared(row)
444  #endif
445  for(IndexType j = i + 1; j < N; ++j) {
446  metric_distance(i, j) = (row - matrix.row(j)).lpNorm<1>();
447  }
448  }
449 }

453 void Clusterator::computeL2Distance(const IncidenceMatrix &matrix) {
454  typedef decltype(matrix.rows()) IndexType;
455 
456  // First, compute the distances.
457  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> XX, D;
458  const auto N = matrix.rows();
459 
460  XX.resize(N, 1);
461  D.resize(N, N);
462 
463  XX = matrix.array().square().rowwise().sum();
464  D.noalias() = XX.replicate(1, N) + XX.transpose().replicate(N, 1);
465  D -= 2 * matrix * matrix.transpose();
466  D = D.cwiseSqrt();
467 
468  // After that, copy to the final destination.
469  metric_distance.resize(N);
470 
471  for(IndexType i = 0; i < D.rows(); ++i)
472  for(IndexType j = 0; j < i + 1; ++j)
473  metric_distance(i, j) = D(i, j);
474 }
