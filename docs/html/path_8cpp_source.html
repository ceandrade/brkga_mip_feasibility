142 void Path::cleanPath(std::string& path) const
143 {
144  if (isEmpty()) return;
145  bool hasRoot = (path[0] == '/');
146  // tokenize
147  std::list<std::string> tokens;
148  typedef boost::tokenizer< boost::char_separator<char> > tokenizer;
149  boost::char_separator<char> sep("/");
150  tokenizer tok(path, sep);
151  for (tokenizer::iterator itr = tok.begin(); itr != tok.end(); ++itr) tokens.push_back(*itr);
152  // sanity check
153  if (hasRoot && tokens.size() && (*tokens.begin()) == std::string("..")) throw std::runtime_error(std::string("Bad path: ") + data);
154  // remove single dots
155  tokens.erase(remove_if(tokens.begin(), tokens.end(), std::bind2nd(std::equal_to<std::string>(), ".")), tokens.end());
156  // remove double dots
157  std::list<std::string>::iterator itr = std::find_if(tokens.begin(), tokens.end(), bind2nd(std::not_equal_to<std::string>(), ".."));
158  std::list<std::string>::iterator end = tokens.end();
159  while (itr != end)
160  {
161  if (*itr == std::string(".."))
162  {
163  std::list<std::string>::iterator tmp = itr++;
164  --tmp;
165  if (*tmp != std::string(".."))
166  {
167  tmp = tokens.erase(tmp);
168  tmp = tokens.erase(tmp);
169  }
170  }

172  }
