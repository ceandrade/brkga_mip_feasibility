218 void PropagatorRounding::apply(const std::vector<double>& in, std::vector<double>& out)
219 {
220  copy(in.begin(), in.end(), out.begin());
221  state->restore();
222  double t = getRoundingThreshold(randomizedRounding, roundGen);
223  ranker->setCurrentState(in);
224  // main loop
225  int next;
226  while ((next = ranker->next()) >= 0)
227  {
228  // standard rounding
229  if (domain.varType(next) == 'B') doRound(in[next], out[next], t);
230  else
231  {
232  // general integer variable: here we take into account also the tightened domain
233  // in order to have a more clever rounding!
234  if (lessEqualThan(in[next], domain.varLb(next))) out[next] = domain.varLb(next);
235  else if (greaterEqualThan(in[next], domain.varUb(next))) out[next] = domain.varUb(next);
236  else doRound(in[next], out[next], t);
237  }
238  // propagate
239  prop.propagate(next, out[next]);
240  DOMINIQS_ASSERT( domain.isVarFixed(next) );
241  // update with fixings
242  for (int j: prop.getLastFixed()) out[j] = domain.varLb(j);
243  }
244 }
