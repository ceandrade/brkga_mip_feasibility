268 Propagator* VarBoundFactory::analyze(Domain* d, Constraint* c)
269 {
270  if ((c->row.size() > 2) || (c->sense == 'E')) return 0;
271  DOMINIQS_ASSERT( (c->sense == 'L') || (c->sense == 'G') );
272  const int* idx = c->row.idx();
273  const double* coef = c->row.coef();
274  // both binaries or both continuous are not allowed
275  int numBin = (d->varType(idx[0]) == 'B') + (d->varType(idx[1]) == 'B');
276  int numCont = (d->varType(idx[0]) == 'C') + (d->varType(idx[1]) == 'C');
277  if ((numBin == 2) || (numCont == 2)) return 0;
278  // map variables to x and y
279  int xIdx = -1;
280  double xCoef = 0.0;
281  int yIdx = -1;
282  double yCoef = 0.0;
283  if (numBin)
284  {
285  if (d->varType(idx[0]) == 'B')
286  {
287  xIdx = idx[1];
288  xCoef = coef[1];
289  yIdx = idx[0];
290  yCoef = coef[0];
291  }
292  else
293  {
294  xIdx = idx[0];
295  xCoef = coef[0];
296  yIdx = idx[1];
297  yCoef = coef[1];
298  }
299  }
300  else
301  {
302  if (d->varType(idx[0]) == 'I')
303  {
304  xIdx = idx[1];
305  xCoef = coef[1];
306  yIdx = idx[0];
307  yCoef = coef[0];
308  }
309  else
310  {
311  xIdx = idx[0];
312  xCoef = coef[0];
313  yIdx = idx[1];
314  yCoef = coef[1];
315  }
316  }
317  // normalize coefficients
318  char sense = c->sense;
319  if (isNegative(xCoef))
320  {
321  // flip sense
322  if (sense == 'L') sense = 'G';
323  else sense = 'L';
324  }
325  xCoef = 1.0;
326  yCoef /= xCoef;
327  double rhs = c->rhs / xCoef;
328  Propagator* prop = 0;
329  if (sense == 'L') prop = new VarUpperBoundProp(d, this, c->name, xIdx, yIdx, yCoef, rhs);
330  else prop = new VarLowerBoundProp(d, this, c->name, xIdx, yIdx, yCoef, rhs);
331  DOMINIQS_ASSERT( prop );
332  numCreated++;
333  return prop;
334 }
