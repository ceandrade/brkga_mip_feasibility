30 static inline double GMIcoef(double alpha, char type, double f0, double ratio, double eps) /* ratio = f0 / (1 - f0) */
31 {
32  if (type != 'C')
33  {
34  // integral
35  double fj = fractionalPart(alpha, eps);
36  return (fj > f0) ? (1.0 - fj) * ratio : fj;
37  }
38  else
39  {
40  // continuous
41  return isNegative(alpha, eps) ? (-alpha * ratio) : alpha;
42  }
43  return 0.0; // never goes here!
44 }

113 bool GomorySeparator::calculateTableauRow(Model& m, const dominiqs::numarray<double>& multipliers)
114 {
115  rank = 0;
116  // handle rhs
117  beta = dotProduct(&multipliers[0], &shiftedRhs[0], m.numRows);
118  // check fractionality of rhs
119  double f = integralityViolation(beta);
120  if (lessThan(f, minViolation)) return false;
121  if (onlyViolated && (badRatio <= onlyViolBadThr))
122  {
123  double f0 = fractionalPart(beta);
124  double ratio = f0 / (1.0 - f0);
125  double violation = f0;
126  double alpha;
127  for (int k = 0; k < numBadX; k++)
128  {
129  alpha = dotProduct(*(m.cols[badIdx[k]]), &multipliers[0]);
130  if (xStatus[badIdx[k]] == 'U') alpha = -alpha;
131  alpha = GMIcoef(alpha, badType[k], f0, ratio, epsBigVars);
132  // if (xStatus[badIdx[k]] == 'U') alpha = -alpha;
133  violation -= alpha * badVal[k];
134  if (violation < minViolation) return false;
135  }
136  for (int k = numBadX; k < numBad; k++)
137  {
138  alpha = multipliers[badIdx[k]];
139  if (m.sense[badIdx[k]] == 'G') alpha = -alpha;
140  alpha = GMIcoef(alpha, badType[k], f0, ratio, epsBigVars);
141  violation -= alpha * badVal[k];
142  if (violation < minViolation) return false;
143  }
144  }
145  // handle structural and artificial variables
146  DOMINIQS_DEBUG_LOG( tabRowChrono.start() );
147  xCoef.resize(m.numVars);
148  xUsed.resize(m.numVars);
149  xUsed.zero();
150  xSupport.reserve(m.numVars);
151  xSupport.clear();
152  sCoef.resize(m.numRows);
153  sSupport.reserve(m.numRows);
154  sSupport.clear();
155  for (unsigned int i = 0; i < m.numRows; i++)
156  {
157  if (isNotNull(multipliers[i], epsBigVars))
158  {
159  rank = std::max(rank, m.rows[i]->rank);
160  const SparseVector& row = m.rows[i]->row;
161  const int* idx = row.idx();
162  const double* coef = row.coef();
163  int kth;
164  for (unsigned int k = 0; k < row.size(); k++)
165  {
166  kth = idx[k];
167  if (!xUsed[kth])
168  {
169  xUsed[kth] = 1;
170  xSupport.push_back_unsafe(kth);
171  xCoef[kth] = 0.0;
172  }
173  xCoef[kth] += (multipliers[i] * coef[k]);
174  }
175  if (m.sense[i] == 'E') continue;
176  if (m.sense[i] == 'G') sCoef[i] = -multipliers[i];
177  else sCoef[i] = multipliers[i];
178  sSupport.push_back_unsafe(i);
179  }
180  }
181  rank += 1;
182  DOMINIQS_DEBUG_LOG( tabRowChrono.stop() );
183 
184  if (rank > maxRank) return false;
185  int maxSize = std::max(int(maxSupport * m.numVars), 100);
186  if (xSupport.size() > (unsigned int)maxSize) return false;
187  return true;
188 }

196 bool GomorySeparator::applyGMI(Model& m)
197 {
198  DOMINIQS_DEBUG_LOG( applyChrono.start() );
199  double f0 = fractionalPart(beta);
200  beta = f0;
201  double ratio = f0 / (1.0 - f0);
202  // derive >= fractional cut
203  // handle structural variables
204  for (int j: xSupport)
205  {
206  // complement phase
207  if (isNull(xCoef[j], epsBigVars))
208  {
209  xCoef[j] = 0.0;
210  continue;
211  }
212  if (xStatus[j] == 'R')
213  {
214  // if we have a free continuous var with non-zero coef we cannot generate a GMI
215  if (m.xType[j] == 'C') return false;
216  // if we have a free integer var with non-integer coef we cannot generate a GMI
217  if (!isInteger(xCoef[j], epsBigVars)) return false;
218  // xCoef[j] = 0.0;
219  }
220  else if (xStatus[j] == 'F')
221  {
222  // fixed variable => simplify
223  xCoef[j] = 0.0; // because we already shifted the bounds!
224  continue;
225  }
226  else if (xStatus[j] == 'U')
227  {
228  // complement
229  xCoef[j] = -xCoef[j];
230  }
231 
232  // GMI formula phase
233  double fj = fractionalPart(xCoef[j], epsBigVars);
234  if (m.xType[j] != 'C')
235  {
236  // integral
237  if (fj > f0) xCoef[j] = (1.0 - fj) * ratio;
238  else xCoef[j] = fj;
239  }
240  else
241  {
242  // continuous
243  if (isNegative(xCoef[j], epsBigVars)) xCoef[j] = -xCoef[j] * ratio;
244  }
245 
246  // undo complement
247  if (xStatus[j] == 'L') beta += (m.xLb[j] * xCoef[j]);
248  else if (xStatus[j] == 'U')
249  {
250  beta -= (m.xUb[j] * xCoef[j]);
251  xCoef[j] = -xCoef[j];
252  }
253  }
254  // handle artificial variables
255  for(int i: sSupport)
256  {
257  double fi = fractionalPart(sCoef[i], epsBigVars);
258  if (m.rows[i]->slackType != 'C')
259  {
260  // integral
261  if (fi > f0) sCoef[i] = (1.0 - fi) * ratio;
262  else sCoef[i] = fi;
263  }
264  else
265  {
266  // continuous
267  if (isNegative(sCoef[i], epsBigVars)) sCoef[i] = -sCoef[i] * ratio;
268  }
269  }
270  DOMINIQS_DEBUG_LOG( applyChrono.stop() );
271  return true;
272 }

314 void GomorySeparator::setX(dominiqs::Model& m, const std::vector<double>& x)
315 {
316  xStar = x;
317  badIdx.clear();
318  badVal.clear();
319  badType.clear();
320  numBadX = 0;
321  numBad = 0;
322  // get current base
323  ws.read(m.env, m.lp);
324  // handle strucutural variable
325  xStatus.resize(m.numVars);
326  for (unsigned int j = 0; j < m.numVars; j++)
327  {
328  if ((m.xSimplexType[j] == 'B') || (m.xSimplexType[j] == 'H'))
329  {
330  double distanceFromLb = x[j] - m.xLb[j] + epsSmallVars;
331  double distanceFromUb = m.xUb[j] - x[j] + epsSmallVars;
332  if (distanceFromLb <= distanceFromUb)
333  {
334  // treat as non-negative variable
335  xStatus[j] = 'L';
336  }
337  else
338  {
339  // treat as non-positive variable
340  xStatus[j] = 'U';
341  }
342  }
343  else if (m.xSimplexType[j] == 'P')
344  {
345  // non-negative variable
346  xStatus[j] = 'L';
347  }
348  else if (m.xSimplexType[j] == 'N')
349  {
350  // non-positive variable
351  xStatus[j] = 'U';
352  }
353  else if (m.xSimplexType[j] == 'R')
354  {
355  // free variable
356  xStatus[j] = 'R';
357  }
358  else if (m.xSimplexType[j] == 'F')
359  {
360  // fixed variable
361  xStatus[j] = 'F';
362  }
363 
364  // is this a bad var?
365  if (ws.cstat[j] != CPX_BASIC)
366  {
367  if ((xStatus[j] == 'L') && greaterThan(xStar[j], m.xLb[j]))
368  {
369  badIdx.push_back(j);
370  badVal.push_back(xStar[j] - m.xLb[j]);
371  badType.push_back(m.xType[j]);
372  }
373  if ((xStatus[j] == 'U') && lessThan(xStar[j], m.xUb[j]))
374  {
375  badIdx.push_back(j);
376  badVal.push_back(m.xUb[j] - xStar[j]);
377  badType.push_back(m.xType[j]);
378  }
379  if (xStatus[j] == 'R')
380  {
381  badIdx.push_back(j);
382  badVal.push_back(xStar[j]);
383  badType.push_back(m.xType[j]);
384  }
385  }
386  }
387  numBadX = badIdx.size();
388  // shifted rhs & sStar
389  shiftedRhs = m.rhs;
390  sStar.clear();
391  for (unsigned int i = 0; i < m.numRows; i++)
392  {
393  const SparseVector& row = m.rows[i]->row;
394  const int* idx = row.idx();
395  const double* coef = row.coef();
396  for (unsigned int k = 0; k < row.size(); k++)
397  {
398  int j = idx[k];
399  double v = coef[k];
400  if (xStatus[j] == 'L') shiftedRhs[i] -= (v * m.xLb[j]);
401  else if (xStatus[j] == 'U') shiftedRhs[i] -= (v * m.xUb[j]);
402  else if (xStatus[j] == 'F') shiftedRhs[i] -= (v * m.xLb[j]);
403  }
404  if (m.sense[i] != 'E')
405  {
406  double slack = dotProduct(row, &xStar[0]) - m.rhs[i];
407  if (m.sense[i] == 'L') slack = -slack;
408  /*if (isNegative(slack))
409  {
410  std::cout << m.rows[i]->name << " has negative slack: " << std::setprecision(10) << slack << std::endl;
411  }
412  DOMINIQS_ASSERT( !isNegative(slack) );*/
413  sStar.push_back(slack);
414  if (isPositive(slack, epsSmallVars))
415  {
416  badIdx.push_back(i);
417  badVal.push_back(slack);
418  badType.push_back(m.rows[i]->slackType);
419  }
420  }
421  }
422  numBad = badIdx.size();
423  badRatio = (double)numBad / (m.numVars + m.numSlacks);
424  DOMINIQS_ASSERT_EQUAL( sStar.size(), m.numSlacks );
425 }

427 dominiqs::CutPtr GomorySeparator::separateOneGMI(dominiqs::Model& m, const dominiqs::numarray<double>& multipliers, bool& violated)
428 {
429  violated = false;
430  DOMINIQS_ASSERT( multipliers.size() >= m.numRows );
431  if (!calculateTableauRow(m, multipliers)) return CutPtr();
432  if (!applyGMI(m)) return CutPtr();
433  backSubstitute(m);
434  numTried++;
435  CutPtr r(new Cut);
436  r->row.reserve(xSupport.size());
437  int maxSize = std::max(int(maxSupport * m.numVars), 100);
438  double violation = beta;
439  for (int j: xSupport)
440  {
441  if (isNotNull(xCoef[j], epsBigVars))
442  {
443  r->row.push_unsafe(j, xCoef[j]);
444  violation -= (xStar[j] * xCoef[j]);
445  if (r->row.size() > (unsigned int)maxSize) return CutPtr();
446  }
447  }
448  if (onlyViolated && (violation < minViolation)) return CutPtr();
449  if (r->row.size() == 0) return CutPtr();
450  r->sense = 'G';
451  r->rhs = beta;
452  r->slackType = 'C';
453  r->removable = true;
454  r->rank = rank;
455  r->digest();
456  if (r->dynamism() > maxDynamism)
457  {
458  numBadDynamism++;
459  return CutPtr();
460  }
461  r->name = (boost::format("gmic_%1%") % cutCounter++).str();
462 #ifdef DOMINIQS_TRACE_CUTS
463  r->mult = multipliers;
464  r->complemented.clear();
465  for (int j: xSupport)
466  {
467  if (xStatus[j] == 'U') r->complemented.push_back(j);
468  }
469 #endif // DOMINIQS_TRACE_CUTS
470  violated = (violation >= minViolation);
471  return r;
472 }

493 bool GomorySeparator::separateGMI(dominiqs::Model& m, const std::vector<double>& x, dominiqs::CutList& cuts)
494 {
495  watch.start();
496  DOMINIQS_DEBUG_LOG( initChrono.start() );
497  if (forceRefactor) refactor(m);
498  dominiqs::numarray<int> bhead(m.numRows);
499  dominiqs::numarray<double> bval(m.numRows);
500  CHECKED_CPX_CALL( CPXgetbhead, m.env, m.lp, &bhead[0], &bval[0] );
501  for (unsigned int i = 0; i < m.numRows; i++)
502  {
503  if ((bhead[i] < 0) || (m.xType[bhead[i]] == 'C')) bval[i] = 0;
504  else bval[i] = integralityViolation(bval[i], epsSmallVars);
505  }
506  dominiqs::numarray<int> perm(m.numRows);
507  permShellSort(bval.c_ptr(), perm.c_ptr(), m.numRows, std::greater<double>());
508  setX(m, x);
509  DOMINIQS_DEBUG_LOG( initChrono.stop() );
510  bool foundViolated = false;
511  for (unsigned int i = 0; i < m.numRows; i++)
512  {
513  if (bval[perm[i]] > minViolation)
514  {
515  bool thisViolated = false;
516  CutPtr newCut = separateOneGMI(m, perm[i], thisViolated);
517  foundViolated |= thisViolated;
518  if (newCut) cuts.push_back(newCut);
519  }
520  else break;
521  if ((int)cuts.size() > maxGMI) break;
522  }
523  watch.stop();
524  return foundViolated;
