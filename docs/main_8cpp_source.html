82 Prob presolve(Env env, Prob lp, MyData& data, const std::string& path)
83 {
84  int objSense = CPXgetobjsen(env, lp);
85  CHECKED_CPX_CALL( CPXpresolve, env, lp, CPX_ALG_NONE );
86  int preStat;
87  CHECKED_CPX_CALL( CPXgetprestat, env, lp, &preStat, 0, 0, 0, 0 );
88  data.hasPresolve = false;
89  if (preStat == 2) // reduced to empty problem
90  {
91  LOG_START_SECTION("presolvedProblem");
92  LOG_ITEM("hasPresolve", "1");
93  LOG_ITEM("emptyProblem", "1");
94  LOG_END_SECTION();
95  LOG_MSG("Too simple: presolved reduced to empty problem!");
96  FREE_PROB( env, lp );
97  return 0;
98  }
99  else if (preStat == 0) // no presolve
100  {
101  data.objSense = 1;
102  LOG_START_SECTION("presolvedProblem");
103  LOG_ITEM("hasPresolve", "0");
104  LOG_ITEM("emptyProblem", "0");
105  LOG_ITEM("offset", "0");
106  LOG_END_SECTION();
107  }
108  else // get presolved problem
109  {
110  data.objSense = objSense;
111  data.hasPresolve = true;
112  // output offset
113  CProb redlp;
114  CHECKED_CPX_CALL( CPXgetredlp, env, lp, &redlp );
115  CHECKED_CPX_CALL( CPXgetobjoffset, env, redlp, &(data.offset) );
116  LOG_START_SECTION("presolvedProblem");
117  LOG_ITEM("hasPresolve", "1");
118  LOG_ITEM("emptyProblem", "0");
119  LOG_ITEM("offset", lexical_cast<std::string>(data.offset));
120  LOG_END_SECTION();
121  }
122  std::string preName = path + "/" + "presolved.mps.gz";
123  if (data.hasPresolve)
124  {
125  CProb redlp;
126  CHECKED_CPX_CALL( CPXgetredlp, env, lp, &redlp );
127  CHECKED_CPX_CALL( CPXwriteprob, env, redlp, preName.c_str(), 0 );
128  }
129  else { CHECKED_CPX_CALL( CPXwriteprob, env, lp, preName.c_str(), 0 ); }
130  DECL_PROB( env, redlp );
131  CHECKED_CPX_CALL( CPXreadcopyprob, env, redlp, preName.c_str(), 0 );
132  if (data.hasPresolve)
133  {
134  LOG_START_SECTION("presolvedProblem");
135  LOG_ITEM("nvars", CPXgetnumcols(env, redlp));
136  LOG_ITEM("binaries", CPXgetnumbin(env, redlp));
137  LOG_ITEM("integers", CPXgetnumint(env, redlp));
138  LOG_ITEM("nrows", CPXgetnumrows(env, redlp));
139  LOG_END_SECTION();
140  }
141  CPXsetdefaults(env);
142  return redlp;
143 }
