135 void LexLessEqualProp::propagate()
136 {
137  dirty = false;
138  if (state == CSTATE_UNKNOWN)
139  {
140  int xA = x[alpha];
141  int yA = y [alpha];
142  double xLB = domain->varLb(xA);
143  double xUB = domain->varUb(xA);
144  double yLB = domain->varLb(yA);
145  double yUB = domain->varUb(yA);
146  if (alpha == beta - 1)
147  {
148  // enforce x_alpha < y_alpha
149  if (greaterEqualThan(xUB, yUB))
150  {
151  // impose x_alpha.ub < y_alpha.ub
152  double newUB = yUB - 1.0;
153  if (lessThan(newUB, xLB)) { state = CSTATE_INFEAS; return; }
154  if (domain->varType(xA) == 'B') domain->fixBinDown(xA);
155  else domain->tightenUb(xA, newUB);
156  }
157  if (lessEqualThan(yLB, xLB))
158  {
159  // impose y_alpha.lb < x_alpha.lb
160  double newLB = xLB + 1.0;
161  if (greaterThan(newLB, yUB)) { state = CSTATE_INFEAS; return; }
162  if (domain->varType(yA) == 'B') domain->fixBinUp(yA);
163  else domain->tightenLb(yA, newLB);
164  }
165  }
166  else if (alpha < (beta - 1))
167  {
168  // enforce x_alpha <= y_alpha
169  if (greaterThan(xUB, yUB))
170  {
171  // impose x_alpha.ub <= y_alpha.ub
172  double newUB = yUB;

174  if (domain->varType(xA) == 'B') domain->fixBinDown(xA);
175  else domain->tightenUb(xA, newUB);

179  // impose y_alpha.lb <= x_alpha.lb

216 void LexLessEqualProp::updateState(int i)
217 {
218  DOMINIQS_ASSERT( (i >= 0) && (i < n) );
219  if ((i < alpha) || (i >= beta)) return;
220  // update alpha or beta
221  if (i == alpha) increaseAlpha();
222  else if (i == beta - 1)
223  {
224  DOMINIQS_ASSERT( (beta > 0) && (beta - 1) < n );
225  if (equal(domain->varLb(x[beta-1]), domain->varUb(y[beta-1])))
226  {
227  beta = i;
228  decreaseBeta();
229  }
230  }
231  else // alpha < i < beta - 1
232  {
233  if (greaterThan(domain->varLb(x[i]), domain->varUb(y[i])))
234  {
235  beta = i;
236  decreaseBeta();
237  }
238  }
239  // detect entailment or failure
240  if (alpha == beta) state = CSTATE_INFEAS;
241  if (alpha == n) state = CSTATE_ENTAILED;
242  if ((alpha == beta - 1) && checkLex(alpha)) state = CSTATE_ENTAILED;
243 }
