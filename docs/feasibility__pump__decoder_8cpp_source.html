526 // Preconditions: most of data structures initialized on init() must be
527 // available.
528 void FeasibilityPump_Decoder::buildConstraint2VariableMatrix() {
529  #ifdef DEBUG
530  cout << "\n------------------------------------------------------"
531  << "\n> Building constraint-to-variable matrix"
532  << "\n> using ";
533  switch(constraint_filtering_type) {
534  case ConstraintFilteringType::ALL:
535  cout << "all constraints";
536  break;
537  case ConstraintFilteringType::ONLY_NONZERO_DUALS:
538  cout << "constraints with non-zero duals";
539  break;
540  case ConstraintFilteringType::NONZERO_DUALS_NONZERO_SLACKS:
541  cout << "constraints with non-zero duals or zero slacks";
542  break;
543  }
544  cout.flush();
545  #endif
546 
547  vector<size_t> ctr_sorting(constraints_per_thread[0].getSize());
548  for(size_t i = 0; i < ctr_sorting.size(); ++i)
549  ctr_sorting[i] = i;
550 
551  if(constraint_filtering_type != ConstraintFilteringType::ALL)
552  sort(ctr_sorting.begin(), ctr_sorting.end(),
553  [&](size_t i, size_t j) {
554  return (fabs(duals[i]) > fabs(duals[j])) ||
555  (fabs(duals[i] - duals[j]) < EPS &&
556  fabs(slacks[i]) < fabs(slacks[j]));
557  });
558 
559  #ifdef FULLDEBUG
560  for(size_t i = 0; i < ctr_sorting.size(); ++i)
561  cout << "\n> Orig: " << constraints_per_thread[0][i].getName()
562  << " | dual: " << duals[i] << " | slack: " << slacks[i]
563  << " \t->\tSorted: " << constraints_per_thread[0][ctr_sorting[i]].getName()
564  << " | dual: " << duals[ctr_sorting[i]]
565  << " | slack: " << slacks[ctr_sorting[i]];
566  cout << endl;
567  #endif
568 
569  // First, we check if we have non-zero duals and change the constraint
570  // filtering if it is the case. This means that if we have no non-zero
571  // dual values, we must change the filtering to consider the slacks.
572  if(constraint_filtering_type == ConstraintFilteringType::ONLY_NONZERO_DUALS &&
573  fabs(duals[ctr_sorting[0]]) < EPS) {
574  constraint_filtering_type = ConstraintFilteringType::NONZERO_DUALS_NONZERO_SLACKS;
575 
576  #ifdef FULLDEBUG
577  cout << "\n** No non-zero dual values are found. Switching to slacks!";
578  #endif
579  }
580 
581  // Remove some constraints from the list according to the filtering type
582  auto it_cut = ctr_sorting.end();
583  if(constraint_filtering_type == ConstraintFilteringType::ONLY_NONZERO_DUALS) {
584  it_cut = find_if(ctr_sorting.begin(), ctr_sorting.end(),
585  [&](size_t i) {
586  return fabs(duals[i]) < EPS;
587  });
588  }
589  else if(constraint_filtering_type == ConstraintFilteringType::NONZERO_DUALS_NONZERO_SLACKS) {
590  it_cut = find_if(ctr_sorting.begin(), ctr_sorting.end(),
591  [&](size_t i) {
592  return fabs(duals[i]) < EPS && fabs(slacks[i]) > EPS;
593  });

647 void FeasibilityPump_Decoder::determineFixingPercentage() {
648  #ifdef DEBUG
649  cout << "\n------------------------------------------------------\n"
650  << "> Trying to determine the variable fixing percentage "
651  << endl;
652  #endif
653 
654  auto &model = model_per_thread[0];
655  auto &binary_variables = binary_variables_per_thread[0];
656  auto &cplex = cplex_per_thread[0];
657  auto &relaxer = relaxer_per_thread[0];
658 
659  // First, we undo the previous variable fixing
660  const auto NUM_BINARIES = binary_variables.getSize();
661 
662  vector<pair<IloNum, size_t>> to_be_fixed(NUM_BINARIES);
663 
664  for(IloInt j = 0; j < NUM_BINARIES; ++j) {
665  IloNum value = full_relaxation_variable_values[binary_variables_indices[j]];
666 
667  switch(var_fixing_type) {
668  case FixingType::MOST_ONES:
669  if(value < 0.0)
670  value = -value;
671  break;
672 
673  case FixingType::MOST_ZEROS:
674  if(value > 0.0)
675  value = -value;
676  break;
677 
678  default:
679  value = fabs(value);
680  if(value > 0.5)
681  value = 1 - value;
682  break;
683  }
684 
685  to_be_fixed[j] = make_pair(value, j);
686  }
687 
688  sort(begin(to_be_fixed), end(to_be_fixed), greater<pair<IloNum, size_t>>());
689 
690  // Now, we start with the all variables in zero, one, or fractional
691  // according to var_fixing_type. If the fixing fails, we reduce the size
692  // for the first half and continue doing it until we get a feasible
693  // fixing. If the fixing falls to one variable only, we set var_fixing
694  // to default values (20% in case of zeros, and 5% in case of ones and
695  // fractionals).
696 
697  // NOTE: note that both MOST_ZEROS and MOST_FRACTIONALS options would
698  // work with "i.first < EPS". Both, for numerical representation reasons,
699  // this note work im my current developing environment. So, I decide
700  // to keep extra code here.
701  if(var_fixing_type == FixingType::MOST_ZEROS) {
702  to_be_fixed.erase(find_if(to_be_fixed.begin(), to_be_fixed.end(),
703  [&](const pair<IloNum, size_t>& i) {
704  return i.first < 0.0;
705  }),
706  to_be_fixed.end());
707  }
708  else if(var_fixing_type == FixingType::MOST_ONES) {
709  to_be_fixed.erase(find_if(to_be_fixed.begin(), to_be_fixed.end(),
710  [&](const pair<IloNum, size_t>& i) {
711  return i.first < 1.0;
712  }),
713  to_be_fixed.end());
714  }
715  else {
716  to_be_fixed.erase(find_if(to_be_fixed.begin(), to_be_fixed.end(),
717  [&](const pair<IloNum, size_t>& i) {
718  return i.first < EPS;
719  }),
720  to_be_fixed.end());
721  }
722 
723  // Now, we try to fix blocks of variables until size the right block size.
724  model.remove(relaxer);
725  cplex.setParam(IloCplex::Param::Threads, num_threads);
726  cplex.setParam(IloCplex::Param::Emphasis::MIP, CPX_MIPEMPHASIS_FEASIBILITY);
727 
728  // Save the current bounds
729  vector<UpperLowerBounds> old_bounds;
730  old_bounds.reserve(NUM_BINARIES);
731  for(IloInt i = 0; i < NUM_BINARIES; ++i)
732  old_bounds.emplace_back(binary_variables[i].getLB(),
733  binary_variables[i].getUB());
734 
735  size_t block_size = to_be_fixed.size();
736  bool found_size = false;
737 
738  while(!found_size && block_size > 2) {
739  #ifdef FULLDEBUG
740  cout << "\n-- Block size: " << block_size << endl;
741  #endif
742 
743  size_t begin = 0;
744  while(begin != to_be_fixed.size()) {
745  size_t end = begin + block_size;
746  if(end > to_be_fixed.size())
747  end = to_be_fixed.size();
748 
749  for(size_t i = begin; i < end; ++i) {
750  auto &var = binary_variables[to_be_fixed[i].second];
751 
752  // If the var is already fixed, skip it.
753  if(fabs(var.getUB() - var.getLB()) > EPS) {
754  // Check the violations. If the variables appears with value 1.0 in
755  // more than 50% of the chromosomes, fix to 1.0. If not, fix to 0.0.
756  const auto value_to_be_fixed =
757  (full_relaxation_variable_values[binary_variables_indices[to_be_fixed[i].second]] > 0.5)? 1.0 : 0.0;
758  var.setBounds(value_to_be_fixed, value_to_be_fixed);
759  }
760  }
761 
762  // Check if this fixing of this block of variables is feasible.
763  // NOTE: Yes, I know that this try/catch construction is horrible
764  // slow! But, this is the unique way, AFAIK, to deal with the
765  // CPLEX presolver.
766  try {
767  cplex.presolve(IloCplex::NoAlg);
768  #ifdef FULLDEBUG
769  cout << "- Success in fixing block " << begin << "-" << end << endl;
770  #endif
771 
772  found_size = true;
773  break;
774  }
775  catch(IloException& e) {
776  #ifdef FULLDEBUG
777  cout << "- Fail fixing block " << begin << "-" << end << endl;
778  #endif
779 
780  // Restore the old bounds and go to the next block.
781  for(size_t i = begin; i < end; ++i) {
782  const auto &index = to_be_fixed[i].second;
783  binary_variables[index].setBounds(old_bounds[index].lb,
784  old_bounds[index].ub);
785  }
786 
787  begin = end;
788  }
789  }
790 
791  if(!found_size)
792  block_size = round(block_size / 2.0);
793  }
794 
795  variable_fixing_percentage = block_size / (double) NUM_BINARIES;
796 
797  #ifdef FULLDEBUG
798  cout << "\n> Block size: " << block_size
799  << "\n> variable_fixing_percentage: " << (variable_fixing_percentage * 100)
800  << endl;
801  #endif
802 
803  if(block_size == 2) {
804  if(var_fixing_type == FixingType::MOST_ZEROS)
805  variable_fixing_percentage = 0.20;
806  else
807  variable_fixing_percentage = 0.05;
808 
809  #ifdef DEBUG
810  cout << "> Block size to small. Changing variable_fixing_percentage to "
811  << (variable_fixing_percentage * 100)
812  << endl;
813  #endif
814  }
815 
816  // Recover original settings
817  model.add(relaxer);
818  cplex.setParam(IloCplex::Param::Threads, 1);
819 
820  for(IloInt i = 0; i < NUM_BINARIES; ++i)
821  binary_variables[i].setBounds(old_bounds[i].lb, old_bounds[i].ub);
822 
823 
824  #ifdef DEBUG
825  cout << "\n------------------------------------------------------" << endl;
826  #endif
827 }

1051  auto &fp_objective = fp_objective_per_thread[0];

1082 
1083  case FixingType::MOST_ZEROS:
1084  value = -histogram[j];
1085  break;

1089  if(value > 0.5)
1090  value = 1 - value;
1091  break;
1092  }
1093

1102  cout << value_index.first << " (" << value_index.second << ") ";
1103  #endif

1106

1112  model.remove(fp_objective);
1113  model.remove(relaxer);

1115 
1116  num_fixings = 0;

1118 // fixDivideAndConquer(0, num_to_fix, to_be_fixed, histogram,

1122

1124  model.add(original_objective);
1125  cplex.setParam(IloCplex::Param::TimeLimit, 10.0);
1126 
1127  #ifdef FULLDEBUG
1128  environment_per_thread[0].setOut(cout);
1129  cplex.setParam(IloCplex::Param::MIP::Display, 4);
1130  #endif
1131 
1132  cplex.solve();
1133 
1134  // If infeasible, unfix vars and return fail.
1135  bool worked = true;
1136  if(cplex.getStatus() == IloAlgorithm::Infeasible) {
1137  for(IloInt j = 0; j < NUM_BINARIES; ++j)
1138  binary_variables[j].setBounds(binary_variables_bounds[j].lb,
1139  binary_variables_bounds[j].ub);
1140  worked = false;
1141  #ifdef DEBUG
1142  cout << "\n* After CPLEX probing: " << cplex.getCplexStatus() << endl;
1143  #endif
1144  }
1145 
1146  // If cplex cannot find anything, fix the vars and return success.
1147  if(cplex.getStatus() == IloAlgorithm::Unknown) { // &&
1148 // cplex.getCplexStatus() == IloCplex::AbortTimeLim) {
1149 
1150  for(int i = 0; i < num_threads; ++i)
1151  for(IloInt j = 0; j < NUM_BINARIES; ++j)
1152  binary_variables_per_thread[i][j]
1153  .setBounds(binary_variables[j].getLB(),
1154  binary_variables[j].getUB());
1155 
1156  for(IloInt j = 0; j < NUM_BINARIES; ++j) {
1157  if(binary_variables[j].getLB() < binary_variables[j].getUB())
1158  fixed_vars[j] = -1;
1159  else if(binary_variables[j].getLB() < EPS)
1160  fixed_vars[j] = 0;
1161  else
1162  fixed_vars[j] = 1;
1163  }
1164 
1165  worked = true;
1166  #ifdef DEBUG
1167  cout << "\n* After CPLEX probing: " << cplex.getCplexStatus() << endl;
1168  #endif
1169  }
1170 
1171  // If we found a feasible solution, yey!!! Just report it to stop.
1172  if(cplex.getStatus() == IloAlgorithm::Feasible ||
1173  cplex.getStatus() == IloAlgorithm::Optimal) {
1174 
1175  cplex.getValues(binary_variables, current_values_per_thread[0]);
1176  for(IloInt i = 0; i < NUM_BINARIES; ++i) {
1177  possible_feasible[i] = current_values_per_thread[0][i];
1178  possible_feasible.rounded[i] = (int) current_values_per_thread[0][i];
1179  }
1180 
1181  possible_feasible.feasibility_pump_value = 0.0;
1182  possible_feasible.fractionality = 0.0;
1183  possible_feasible.num_non_integral_vars = 0;
1184  worked = true;
1185  #ifdef DEBUG
1186  cout << "\n* After CPLEX probing: " << cplex.getStatus() << "!!!" << endl;
1187  #endif
1188  }
1189 
1190  // Add back the FP obj. function and the relaxation transformation.
1191  model.remove(original_objective);
1192  model.add(fp_objective);
1193  model.add(relaxer);
1194 
1195  cplex.setParam(IloCplex::Param::TimeLimit, 1e+75);
1196  cplex.setParam(IloCplex::Param::Threads, 1);
1197 
1198  #ifdef DEBUG
1199  cplex.setParam(IloCplex::Param::MIP::Display, 0);
1200  environment_per_thread[0].setOut(environment_per_thread[0].getNullStream());
1201  cout << "\n--------------------------------\n" << endl;
1202  #endif
1203  return worked;
1204 }

1569  auto &relaxer = relaxer_per_thread[0];
1570  auto &original_objective = original_objective_per_thread[0];
1571  auto &fp_objective = fp_objective_per_thread[0];

1574  size_t num_fixed_vars = 0;

1577  vector<int> histogram(NUM_BINARIES, 0);
1578

1580  vector<int8_t> local_fixed(NUM_BINARIES, -1);
1581 
1582  for(unsigned i = 0; i < num_chromosomes; ++i) {
1583  auto &chr = population.getChromosome(i);
1584  for(IloInt j = 0; j < NUM_BINARIES; ++j)

1589  auto value = histogram[var_idx] / (double) num_chromosomes;
1590  if(value < (discrepancy_level + EPS) || value > (1- discrepancy_level - EPS)) {
1591  value = round(value);
1592  binary_variables[var_idx].setBounds(value, value);
1593  local_fixed[var_idx] = int8_t(value);
1594  ++num_fixed_vars;
1595  }
1596  else
1597  binary_variables[var_idx].setBounds(binary_variables_bounds[var_idx].lb,
1598  binary_variables_bounds[var_idx].ub);
1599  }
1600

1602 
1603  #ifdef DEBUG
1604  cout << "\n** After fixing..."
1605  << "\n** Num of fixed variables: "
1606  << num_fixed_vars << " / " << NUM_BINARIES
1607  << " (" << ((100.0 * num_fixed_vars) / NUM_BINARIES)
1608  << "%)"
1609  << "\n** Num of free variables: "
1610  << num_unfixed_vars << " / " << NUM_BINARIES
1611  << " (" << ((100.0 * num_unfixed_vars) / NUM_BINARIES)
1612  << "%)\n"
1613  << endl;
1614  #endif

1617  // Checking violated constraints.
1619 
1620  for(IloInt i = 0; i < constraints.getSize(); ++i) {
1621  if(ExecutionStopper::mustStop())
1622  break;
1623 
1624  auto &ctr = constraints[i];
1625  IloNum fixed_contribution = 0.0;
1626  IloNum positive_contribution = 0.0;
1627  IloNum negative_contribution = 0.0;
1628 
1629  size_t hash_value = 0; // Used to identify cut already taken
1630 
1631  for(auto it = ctr.getLinearIterator(); it.ok(); ++it) {
1632  auto var = it.getVar();
1633  auto value = it.getCoef();
1634 
1635  if(var.getType() == IloNumVar::Bool) {
1636  const auto var_index = binary_variables_id_index[var.getId()];
1637 
1638  if(local_fixed[var_index] == 1) {
1639  fixed_contribution += value;
1640  hash_value ^= (size_t)var_index + 0x9e3779b9 +
1641  (hash_value << 6) + (hash_value >> 2);
1642  }
1643  else
1644  if(local_fixed[var_index] == -1) {
1645  if(value > 0)
1646  positive_contribution += value;
1647  else
1648  negative_contribution += value;
1649  }
1650  }
1651  else {
1652  if(value > 0)
1653  positive_contribution += value;
1654  else
1655  negative_contribution += value;
1656  }
1657  }

1660  char constraint_type;
1661 
1662  // <= inequalities
1663  if(ctr.getLB() == -IloInfinity && ctr.getUB() < IloInfinity) {

1666  // >= inequalities
1667  else if(ctr.getLB() > -IloInfinity && ctr.getUB() == IloInfinity) {
1668  constraint_type = 'g';
1669  }

1671  else if(fabs(ctr.getUB() - ctr.getLB()) < EPS) {
1672  constraint_type = 'e';

1684  (fixed_contribution + negative_contribution > ctr.getUB());

1687  if(!violated && constraint_type == 'g')
1688  violated = fixed_contribution + positive_contribution < ctr.getLB();
1689

1691  if(!violated && constraint_type == 'e') {
1692  auto surplus = fixed_contribution - ctr.getUB();

1750  (local_fixed[binary_variables_id_index[var.getId()]] == 0 &&

1765  << "\n** Num of fixed variables: "
1766  << num_fixed_vars << " / " << NUM_BINARIES

1810 
1811  // First, we insert a general cut.
1812  IloExpr expr(env);
1813  int accum = 0;
1814  size_t hash_value = 0;
1815 
1816  for(IloInt i = 0; i < NUM_BINARIES; ++i) {
1817  if(local_fixed[i] == 0) {
1818  expr -= binary_variables[i];
1819  }
1820  else if(local_fixed[i] == 1) {
