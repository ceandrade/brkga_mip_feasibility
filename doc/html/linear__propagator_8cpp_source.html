310  double newB;

312  // update maxActDelta if necessary
313  if (maxActDelta < 0.0)
314  {
315  for (k = 0; k < lastPosBin; k++)
316  {
317  if (domain->isVarFixed(posBinIdx[k])) continue;
318  if (maxActDelta < posBinCoef[k]) { maxActDelta = posBinCoef[k]; }
319  }
320  for (k = 0; k < lastNegBin; k++)
321  {
322  if (domain->isVarFixed(negBinIdx[k])) continue;
323  if (maxActDelta < -negBinCoef[k]) { maxActDelta = -negBinCoef[k]; }
324  }
325  for (k = 0; k < lastPos; k++)
326  {
327  if (domain->isVarFixed(posIdx[k])) continue;
328  double alpha = posCoef[k] * (domain->varUb(posIdx[k]) - domain->varLb(posIdx[k]));
329  if (maxActDelta < alpha) { maxActDelta = alpha; }
330  }
331  for (k = 0; k < lastNeg; k++)
332  {
333  if (domain->isVarFixed(negIdx[k])) continue;
334  double alpha = negCoef[k] * (domain->varLb(negIdx[k]) - domain->varUb(negIdx[k]));
335  if (maxActDelta < alpha) { maxActDelta = alpha; }
336  }
337  }
338  if (maxActDelta < 0.0)
339  {
340  // may happen if all variables are fixed
341  updateState();
342  return;
343  }
344  double slack = lessThan(rhs, INFBOUND) ? rhs - minAct : INFBOUND;
345  double surplus = greaterThan(lhs, -INFBOUND) ? maxAct - lhs : INFBOUND;
346  if (lessEqualThan(maxActDelta, std::min(slack, surplus))) return; //< cannot deduce anything!
347  maxActDelta = -1.0; //< will be recalculated on the next propagation!
348  // one pass
349  if (lessThan(rhs, INFBOUND))
350  {
351  double beta = rhs - minAct;
352  if (minActInfCnt == 0)
353  {
354  for (k = 0; (k < lastPosBin) && (state == CSTATE_UNKNOWN); k++)

358  a = posBinCoef[k];
359  if (greaterThan(a, beta)) { domain->fixBinDown(j); factory->domainReductions()++; }
360  }

363  j = negBinIdx[k];

365  a = negBinCoef[k];
366  if (greaterThan(-a, beta)) { domain->fixBinUp(j); factory->domainReductions()++; }
367  }
368  for (k = 0; (k < lastPos) && (state == CSTATE_UNKNOWN); k++)
369  {
370  j = posIdx[k];
371  if (domain->isVarFixed(j)) continue;
372  a = posCoef[k];
373  double alpha = a * (domain->varUb(j) - domain->varLb(j));
374  if (greaterThan(alpha, beta))
375  {
376  newB = domain->varLb(j) + beta / a;
377  if (domain->varType(j) != 'C') newB = floorEps(newB);
378  domain->tightenUb(j, newB);
379  factory->domainReductions()++;
380  }
381  }

404  }

785 class KnapsackAdvisor : public AdvisorI
786 {
787 public:
788  KnapsackAdvisor(KnapsackProp* p, int j, double coef) : AdvisorI(p, j), a(coef) {}
789  // events for binary variables
790  void fixedUp()
791  {
792  if (prop->getState() != CSTATE_UNKNOWN) return;
793  KnapsackProp* p = getMyProp<KnapsackProp>();
794  p->minAct += a;
795  p->dirty |= lessThan(p->rhs, INFBOUND);
796  }
797  void fixedDown()
798  {
799  if (prop->getState() != CSTATE_UNKNOWN) return;
800  KnapsackProp* p = getMyProp<KnapsackProp>();
801  p->maxAct -= a;
802  p->dirty |= greaterThan(p->lhs, -INFBOUND);
803  }
804  // events for other variables
805  void tightenLb(double delta, bool decreaseInfCnt, bool propagate)
806  {
807  if (prop->getState() != CSTATE_UNKNOWN) return;
808  KnapsackProp* p = getMyProp<KnapsackProp>();
809  p->minAct += delta * a;
810  p->dirty |= (propagate && lessThan(p->rhs, INFBOUND));
811  }
812  void tightenUb(double delta, bool decreaseInfCnt, bool propagate)
813  {
814  if (prop->getState() != CSTATE_UNKNOWN) return;
815  KnapsackProp* p = getMyProp<KnapsackProp>();
816  p->maxAct += delta * a;
817  p->dirty |= (propagate && greaterThan(p->lhs, -INFBOUND));
818  }
819  // output
820  std::ostream& print(std::ostream& out) const
821  {
822  return out << format("adv(%1%, idx=%2% coef=%3%)") % prop->getName() % var % a;
823  }
824 protected:
825  double a;
826 };

828 KnapsackProp::KnapsackProp(Domain* d, PropagatorFactory* fact, Constraint* c) : Propagator(d, fact)
829 {
830  DOMINIQS_ASSERT( domain );
831  DOMINIQS_ASSERT( c );
832  name = c->name;
833  // assuming data is correct
834  switch(c->sense)
835  {
836  case 'L':
837  lhs = -INFBOUND;
838  rhs = c->rhs;
839  break;
840  case 'E':
841  lhs = c->rhs;
842  rhs = c->rhs;
843  break;
844  case 'G':
845  lhs = c->rhs;
846  rhs = INFBOUND;
847  break;
848  default:
849  throw std::runtime_error("Unknown constraint sense!");
850  }
851  minAct = 0.0;
852  maxAct = 0.0;
853  unsigned int rowSize = c->row.size();
854  const int* idx = c->row.idx();
855  const double* coef = c->row.coef();
856  for (unsigned int k = 0; k < rowSize; k++)
857  {
858  int j = idx[k];
859  double a = coef[k];
860  if (isNull(a)) continue;
861  DOMINIQS_ASSERT( a > 0.0 );
862  if (domain->varType(j) == 'B')
863  {
864  maxAct += a;
865  posBinIdx.push_back(j);
866  posBinCoef.push_back(a);
867  }
868  else
869  {
870  minAct += a * domain->varLb(j);
871  maxAct += a * domain->varUb(j);
872  posIdx.push_back(j);
873  posCoef.push_back(a);
874  }
875  }
876  lastPosBin = posBinIdx.size();
877  lastPos = posIdx.size();
878  maxActDelta = -1.0;
879  updateState();
880  setPriority(KNAPSACK_DEFAULT_PRIORITY);
881 }

902 void KnapsackProp::updateState()
903 {
904  if (state != CSTATE_UNKNOWN) return;
905  if (greaterThan(minAct, rhs) || lessThan(maxAct, lhs))
906  {
907  state = CSTATE_INFEAS;
908  dirty = false;
909  return;
910  }
911  if (lessEqualThan(maxAct, rhs) && greaterEqualThan(minAct, lhs))
912  {
913  state = CSTATE_ENTAILED;
914  dirty = false;
915  return;
916  }

919 void KnapsackProp::propagate()
920 {
921  updateState();
922  if (dirty == false) return;
923  factory->propCalled()++;
924  unsigned int k;
925  int j;
926  double a;
927  double newB;
928  dirty = false; // here at beginning because this is not monotonic
929  // update maxActDelta if necessary
930  if (maxActDelta < 0.0)
931  {
932  for (k = 0; k < lastPosBin; k++)
933  {
934  if (domain->isVarFixed(posBinIdx[k])) continue;
935  if (maxActDelta < posBinCoef[k]) { maxActDelta = posBinCoef[k]; }
936  }
937  for (k = 0; k < lastPos; k++)
938  {
939  if (domain->isVarFixed(posIdx[k])) continue;
940  double alpha = posCoef[k] * (domain->varUb(posIdx[k]) - domain->varLb(posIdx[k]));
941  if (maxActDelta < alpha) { maxActDelta = alpha; }
942  }
943  }
944  if (maxActDelta < 0.0)
945  {
946  // may happen if all variables are fixed
947  updateState();
948  return;
949  }
950  double slack = lessThan(rhs, INFBOUND) ? (rhs - minAct) : INFBOUND;
951  double surplus = greaterThan(lhs, -INFBOUND) ? (maxAct - lhs) : INFBOUND;
952  if (lessEqualThan(maxActDelta, std::min(slack, surplus))) return; //< cannot deduce anything!
953  maxActDelta = -1.0; //< will be recalculated on the next propagation!
954  // one pass
955  if (lessThan(rhs, INFBOUND))
956  {
957  double beta = rhs - minAct;
958  for (k = 0; (k < lastPosBin) && (state == CSTATE_UNKNOWN); k++)
959  {
960  j = posBinIdx[k];
961  if (domain->isVarFixed(j)) continue;
962  a = posBinCoef[k];
963  if (greaterThan(a, beta)) { domain->fixBinDown(j); factory->domainReductions()++; }
964  }
965  for (k = 0; (k < lastPos) && (state == CSTATE_UNKNOWN); k++)
966  {
967  j = posIdx[k];
968  if (domain->isVarFixed(j)) continue;
969  a = posCoef[k];
970  double alpha = a * (domain->varUb(j) - domain->varLb(j));
971  if (greaterThan(alpha, beta))
972  {
973  newB = domain->varLb(j) + beta / a;
974  if (domain->varType(j) != 'C') newB = floorEps(newB);
975  domain->tightenUb(j, newB);
976  factory->domainReductions()++;
977  }
978  }
979  }
980  if (greaterThan(lhs, -INFBOUND))
981  {
982  double beta = maxAct - lhs;
983  for (k = 0; (k < lastPosBin) && (state == CSTATE_UNKNOWN); k++)
984  {
985  j = posBinIdx[k];
986  if (domain->isVarFixed(j)) continue;
987  a = posBinCoef[k];
988  if (greaterThan(a, beta)) { domain->fixBinUp(j); factory->domainReductions()++; }
989  }
990  for (k = 0; (k < lastPos) && (state == CSTATE_UNKNOWN); k++)
991  {
992  j = posIdx[k];
993  if (domain->isVarFixed(j)) continue;
994  a = posCoef[k];
995  double alpha = a * (domain->varUb(j) - domain->varLb(j));
996  if (greaterThan(alpha, beta))
997  {
998  newB = domain->varUb(j) - beta / a;
999  if (domain->varType(j) != 'C') newB = ceilEps(newB);
1000  domain->tightenLb(j, newB);
1001  factory->domainReductions()++;
1002  }
1003  }
1004  }
1005 }
