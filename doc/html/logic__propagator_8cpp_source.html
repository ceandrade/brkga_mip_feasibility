293 Propagator* LogicFactory::analyze(Domain* d, Constraint* c)
294 {
295  bool isImplies = (c->row.size() == 2) && (c->sense != 'G') && isNull(c->rhs);
296  const int* idx = c->row.idx();
297  const double* coef = c->row.coef();
298  if (isImplies)
299  {
300  if (d->varType(idx[0]) != 'B') return 0;
301  if (d->varType(idx[1]) != 'B') return 0;
302  double minCoef = std::min(coef[0], coef[1]);
303  double maxCoef = std::max(coef[0], coef[1]);
304  if (equal(minCoef, -1.0) && equal(maxCoef, 1.0))
305  {
306  numCreated++;
307  if (c->sense == 'L')
308  {
309  if (coef[0] > 0) return new ImpliesProp(d, this, c->name, idx[0], idx[1]);
310  else return new ImpliesProp(d, this, c->name, idx[1], idx[0]);
311  }
312  else if (c->sense == 'E') return new EquivProp(d, this, c->name, idx[0], idx[1]);
313  }
314  }
315  return 0;
316 }
