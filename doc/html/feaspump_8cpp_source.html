434 void FeasibilityPump::perturbe(std::vector<double>& x, bool ignoreGeneralIntegers)
435 {
436  MYLOG( logPerturbation, gLog().startSection("perturbe"); );
437  pertCnt++;
438  // order
439  std::multimap< double, int, std::greater<double> > toOrder;
440  double sigma;
441  if (ignoreGeneralIntegers)
442  {
443  for (int j: binaries)
444  {
445  if (perturbeFromTransformerOutput) sigma = fabs(x[j] - frac_x[j]);
446  else sigma = fabs(floor(frac_x[j] + 0.5) - frac_x[j]);
447  if (sigma > sigmaThr) toOrder.insert(std::pair<double, int>(sigma, j));
448  }
449  }
450  else
451  {
452  for (int j: integers)
453  {
454  if (perturbeFromTransformerOutput) sigma = fabs(x[j] - frac_x[j]);
455  else sigma = fabs(floor(frac_x[j] + 0.5) - frac_x[j]);
456  if (sigma > sigmaThr) toOrder.insert(std::pair<double, int>(sigma, j));
457  }
458  }
459  if (toOrder.empty()) return;
460 
461  // flip
462  int nflips = avgFlips * (rnd.getFloat() + 0.5);
463  int flipsDone = 0;
464  std::multimap<double, int>::const_iterator itr = toOrder.begin();
465  std::multimap<double, int>::const_iterator end = toOrder.end();
466  while ((itr != end) && (flipsDone < nflips))
467  {
468  int toFlip = itr->second;
469  if (equal(x[toFlip], model.xLb[toFlip], integralityEps)) { x[toFlip] += 1.0; ++flipsDone; }
470  else if (equal(x[toFlip], model.xUb[toFlip], integralityEps)) { x[toFlip] -= 1.0; ++flipsDone; }
471  else
472  {
473  // this can happen only with general integer variables
474  if (lessThan(x[toFlip], frac_x[toFlip], integralityEps)) { x[toFlip] += 1.0; ++flipsDone; }
475  if (greaterThan(x[toFlip], frac_x[toFlip], integralityEps)) { x[toFlip] -= 1.0; ++flipsDone; }
476  }
477  ++itr;
478  }
479  DOMINIQS_ASSERT( flipsDone );
480  display.set("P", IT_DISPLAY_SIMPLE_FMT(char, '*'));
481  display.set("#flips", IT_DISPLAY_SIMPLE_FMT(int, flipsDone));
482  MYLOG( logPerturbation,
483  gLog().logItem("maxFlips", nflips);
484  gLog().logItem("flipsDone", flipsDone);
485  gLog().endSection();
486  );
487 }

489 void FeasibilityPump::restart(std::vector<double>& x, bool ignoreGeneralIntegers)
490 {
491  MYLOG( logPerturbation, gLog().startSection("restart"); );
492  restartCnt++;
493  // get previous solution
494  DOMINIQS_ASSERT( lastIntegerX.size() );
495  const std::vector<double>& previousSol = lastIntegerX.begin()->second;
496  // perturbe
497  double sigma;
498  double r;
499  // perturbe binaries
500  int changed = 0;
501  unsigned int size = binaries.size();
502  for (unsigned int i = 0; i < size; i++)
503  {
504  int j = binaries[i];
505  r = rnd.getFloat() - 0.47;
506  if ( r > 0 && equal(x[j], previousSol[j], integralityEps) )
507  {
508  if (perturbeFromTransformerOutput) sigma = fabs(x[j] - frac_x[j]);
509  else sigma = fabs(floor(frac_x[j] + 0.5) - frac_x[j]);
510  if ( (sigma + r) > 0.5 )
511  {
512  x[j] = isNull(x[j], integralityEps) ? 1.0 : 0.0;
513  ++changed;
514  }
515  }
516  }
517 
518  if (!ignoreGeneralIntegers && gintegers.size())
519  {
520  if (lastRestart < nitr)
521  {
522  while (lastRestart++ < nitr) flipsInRestart *= GEOM_FACTOR; // geometric decrease
523  }
524  flipsInRestart = std::min(flipsInRestart + 2 * avgFlips + 1, maxFlipsInRestart); // linear increase
525  DOMINIQS_ASSERT( flipsInRestart );
526  // perturbe general integers
527  double newValue;
528  for (int i = 0; i < flipsInRestart; i++)
529  {
530  int randIdx = int(rnd.getFloat() * (gintegers.size() - 1));
531  DOMINIQS_ASSERT( randIdx >= 0 && randIdx < (int)gintegers.size() );
532  int j = gintegers[randIdx];
533  double lb = model.xLb[i];
534  double ub = model.xUb[i];
535  r = rnd.getFloat();
536  if( (ub - lb) < BIGBIGM ) newValue = floor(lb + (1 + ub - lb) * r);
537  else if( (x[j] - lb) < BIGM ) newValue = lb + (2 * BIGM - 1) * r;
538  else if( (ub - x[j]) < BIGM ) newValue = ub - (2 * BIGM - 1) * r;
539  else newValue = x[j] + (2 * BIGM - 1) * r - BIGM;
540  newValue = std::max(std::min(newValue, ub), lb);
541  if( different(newValue, x[j], integralityEps) )
542  {
543  x[j] = newValue;
544  ++changed;
545  }
546  }
547  DOMINIQS_ASSERT( changed );
548  }
549  MYLOG( logPerturbation,
550  gLog().logItem("flipsInRestart", flipsInRestart);
551  gLog().logItem("flipsDone", changed);
552  gLog().endSection();
553  );
