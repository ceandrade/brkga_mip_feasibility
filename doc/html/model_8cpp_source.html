219 {
220  DOMINIQS_ASSERT( env && lp );
221  int deleted = 0;
222  std::vector<int> delStat(numRows);
223  // update age and decide which are the cuts to remove
224  for (unsigned int i = numOrigRows; i < numRows; i++)
225  {
226  if (rows[i]->isSlack(&x[0], linearEPS) && rows[i]->removable) rows[i]->age++;
227  else rows[i]->age = 0;
228  if (rows[i]->age > ageLimit)
229  {
230  delStat[i] = 1;
231  deleted++;
232  }
233  }
234  // remove cuts from Cplex
235  CHECKED_CPX_CALL( CPXdelsetrows, env, lp, &delStat[0] );

237  // remove cuts from Model too
238  for (unsigned int i = numOrigRows; i < numRows; i++)
239  {

241  else rows[delStat[i]] = rows[i];

252  for (unsigned int i = 0; i < numRows; i++) if (rows[i]->sense != 'E') numSlacks++;
253  }

348 int Model::reducedCostFixing(double primalBound)
349 {
350  DOMINIQS_ASSERT( env && lp );
351  double dualBound = getObjVal();
352  dominiqs::numarray<double> red(numVars);
353  CHECKED_CPX_CALL( CPXgetdj, env, lp, &red[0], 0, numVars - 1 );
354  dominiqs::numarray<int> cstat(numVars);
355  dominiqs::numarray<int> rstat(numRows);
356  CHECKED_CPX_CALL( CPXgetbase, env, lp, &cstat[0], &rstat[0] );
357 
358  int fixed = 0;
359  char change = 'B';
360  double delta = fabs(primalBound - dualBound);
361  for (unsigned int j = 0; j < numVars; j++)
362  {
363  if ((cstat[j] != CPX_BASIC) && different(xLb[j], xUb[j]) && greaterThan(fabs(red[j]), delta))
364  {
365  int idx = (int)j;
366  double val = (cstat[j] == CPX_AT_LOWER) ? xLb[j] : xUb[j];
367  CHECKED_CPX_CALL( CPXchgbds, env, lp, 1, &idx, &change, &val );
368  xLb[j] = xUb[j] = val;
369  fixed++;
370  }
371  }
372  return fixed;
373 }
