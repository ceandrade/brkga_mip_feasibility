60 void FeasibilityPump_Decoder::roundingWithConstraintPropagation(
61  const IloNumArray &in, IloNumArray &out) {
62  #ifdef DEBUG
63  cout << "\n--------------------------------\n"
64  << "> Rounding with constraint propagation "
65  << endl;
66  #endif
67 
68  #ifdef _OPENMP
69  vector<double> &frac = frac_fp_per_thread[omp_get_thread_num()];
70  vector<double> &rounded = rounded_fp_per_thread[omp_get_thread_num()];
71  dominiqs::SolutionTransformerPtr &frac2int = frac2int_per_thread[omp_get_thread_num()];
72  #else
73  vector<double> &frac = frac_fp_per_thread[0];
74  vector<double> &rounded = rounded_fp_per_thread[0];
75  dominiqs::SolutionTransformerPtr &frac2int = frac2int_per_thread[0];
76  #endif
77 
78  // Copy the fractional values for binary vars.
79  for(IloInt i = 0; i < in.getSize(); ++i)
80  frac[binary_variables_indices[i]] = in[i];
81 
82  // Round.
83  frac2int->apply(frac, rounded);
84 
85  // Copy back.
86  for(IloInt i = 0; i < in.getSize(); ++i)
87  out[i] = rounded[binary_variables_indices[i]];
88 
89  #ifdef FULLDEBUG
90  cout << "Frac Rounded";
91  for(size_t i = 0; i < rounded.size(); ++i)
92  cout << "\n" << setiosflags(ios::fixed) << setprecision(2)
93  << frac[i] << " - " << rounded[i];
94  cout << endl;
95  #endif
96 }
