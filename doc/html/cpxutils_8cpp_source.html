398  // extract row info

400  std::vector<double> rhs(rows);

402  CHECKED_CPX_CALL( CPXgetsense, env, lp, &sense[0], 0, rows - 1 );

407  std::vector<int> matind(numNz);

411  ConstraintPtr c(new Constraint);
