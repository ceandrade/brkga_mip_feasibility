127 bool IniFile::load(const std::string& filename)
128 {
129  IniParserContext ctx(this);
130 
131  // GRAMMAR
132 
133  // helper rules
134  rule_t char_ident_start = alpha_p | ch_p('_') ;
135  rule_t char_ident_middle = alnum_p | ch_p('_') ;
136  rule_t ident = char_ident_start >> * char_ident_middle ;
137  rule_t char_start_comment = ch_p('#') | ch_p(';') | str_p("//");
138  rule_t blanks_p = * blank_p;
139  rule_t value_p = * ( alnum_p | blank_p | punct_p );
140 
141  // empty lines and comments
142  rule_t l_comment = blanks_p >> char_start_comment >> * print_p >> eol_p;
143  rule_t l_empty = blanks_p >> eol_p;
144  rule_t c_comment_rule = confix_p("/*", *anychar_p, "*/");
145  rule_t b_comment =
146  blanks_p >>
147  c_comment_rule >>
148  blanks_p >>
149  eol_p;
150 
151  // category
152  rule_t l_category =
153  blanks_p >>
154  ch_p('[') >>
155  blanks_p >>
156  ident [ boost::bind(&IniParserContext::addCategory, &ctx, _1, _2) ] >>
157  blanks_p >>
158  ch_p(']') >>
159  blanks_p >>
160  eol_p;
161 
162  // entry
163  rule_t l_entry =
164  blanks_p >>
165  ident [ boost::bind(&IniParserContext::addEntry, &ctx, _1, _2) ] >>
166  blanks_p >>
167  ch_p('=') >>
168  blanks_p >>
169  value_p [ boost::bind(&IniParserContext::addValue, &ctx, _1, _2) ] >>
170  blanks_p >>
171  eol_p;
172 
173  // whole file
174  rule_t lines = l_comment | b_comment | l_category | l_entry | l_empty;
175  rule_t ini_file = lexeme_d [ * lines ] ;
176 
177  // Create a file iterator for this file
178  iterator_t first(filename.c_str());
179  if (!first) return false;
180  // Create an EOF iterator
181  iterator_t last = first.make_end();
182  int errcode = parse(first, last, ini_file).full;
183  return errcode;
184 }
