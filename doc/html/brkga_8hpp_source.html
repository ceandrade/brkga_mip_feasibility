282 BRKGA<Decoder, RNG>::BRKGA(unsigned _n, unsigned _p, double _pe, double _pm, double _rhoe,
283  Decoder& decoder, RNG& rng, unsigned _K, Sense sense, unsigned MAX, Allele _left_lb,
284  Allele _left_ub, unsigned _cut_point,
285  Allele _right_lb, Allele _right_ub):
286  n(_n), p(_p),
287  pe(unsigned(_pe * p)), pm(unsigned(_pm * p)), rhoe(_rhoe),
288  refRNG(rng), refDecoder(decoder), K(_K), MAX_THREADS(MAX),
289  left_lb(_left_lb), left_ub(_left_ub), cut_point(_cut_point),
290  right_lb(_right_lb), right_ub(_right_ub), previous(K, 0),
291  current(K, 0), initialPopulation(false), initialized(false),
292  reset_phase(false),
293  maximize(sense == Sense::MAXIMIZE)
294 {
295  // Error check:
296  using std::range_error;
297  if(n == 0) { throw range_error("Chromosome size equals zero."); }
298  if(p == 0) { throw range_error("Population size equals zero."); }
299  if(pe == 0) { throw range_error("Elite-set size equals zero."); }
300  if(pe > p) { throw range_error("Elite-set size greater than population size (pe > p)."); }
301  if(pm > p) { throw range_error("Mutant-set size (pm) greater than population size (p)."); }
302  if(pe + pm > p) { throw range_error("elite + mutant sets greater than population size (p)."); }
303  if(K == 0) { throw range_error("Number of parallel populations cannot be zero."); }
304 
305  if(cut_point == 0) {
306  right_lb = left_lb;
307  right_ub = left_ub;
308  }

353 template<class Decoder, class RNG>
354 void BRKGA<Decoder, RNG>::evolve(unsigned generations) {
355  if(!initialized) {
356  throw std::runtime_error("The algorithm hasn't been initialized. Don't forget to call initialize() method");
357  }
358 
359  if(generations == 0) { throw std::range_error("Cannot evolve for 0 generations."); }
360 
361  for(unsigned i = 0; i < generations; ++i) {
362  for(unsigned j = 0; j < K; ++j) {
363  evolution(*current[j], *previous[j]); // First evolve the population (curr, next)
364  std::swap(current[j], previous[j]); // Update (prev = curr; curr = prev == next)
365  }
366  }

369 template<class Decoder, class RNG>
370 void BRKGA<Decoder, RNG>::exchangeElite(unsigned M) {
371  if(M == 0 || M >= p) { throw std::range_error("M cannot be zero or >= p."); }
372 
373  #ifdef _OPENMP
374  #pragma omp parallel for num_threads(MAX_THREADS)
375  #endif
376  for(int i = 0; i < int(K); ++i) {
377  // Population i will receive some elite members from each Population j below:
378  unsigned dest = p - 1; // Last chromosome of i (will be overwritten below)
379  for(unsigned j = 0; j < K; ++j) {
380  if(j == unsigned(i)) { continue; }
381 
382  // Copy the M best of Population j into Population i:
383  for(unsigned m = 0; m < M; ++m) {
384  // Copy the m-th best of Population j into the 'dest'-th position of Population i:
385  const Chromosome& bestOfJ = current[j]->getChromosome(m);
386 
387  std::copy(bestOfJ.begin(), bestOfJ.end(), current[i]->getChromosome(dest).begin());
388  current[i]->fitness[dest].first = current[j]->fitness[m].first;
389 
390  --dest;
391  }
392  }
393  }
394 
395  // PARALLEL REGION: re-sort each population since they were modified
396  #ifdef _OPENMP
397  #pragma omp parallel for num_threads(MAX_THREADS)
398  #endif
399  for(int i = 0; i < int(K); ++i) { current[i]->sortFitness(maximize); }
400 }

505 template<class Decoder, class RNG>
506 inline void BRKGA<Decoder, RNG>::evolution(Population& curr, Population& next) {
507  // We now will set every chromosome of 'current', iterating with 'i':
508  unsigned i = 0; // Iterate chromosome by chromosome
509  unsigned j = 0; // Iterate allele by allele
510 
511  // 2. The 'pe' best chromosomes are maintained, so we just copy these into 'current':
512  while(i < pe) {
513 // for(j = 0 ; j < n; ++j) { next(i,j) = curr(curr.fitness[i].second, j); }
514  next.population[i] = curr.population[curr.fitness[i].second];
515 
516  next.fitness[i].first = curr.fitness[i].first;
517  next.fitness[i].second = i;
518  ++i;
519  }
520 
521  // 3. We'll mate 'p - pe - pm' pairs; initially, i = pe, so we need to iterate until i < p - pm:
522  while(i < p - pm) {
523  // Select an elite parent:
524  const unsigned eliteParent = (refRNG.randInt(pe - 1));
525 
526  // Select a non-elite parent:
527  const unsigned noneliteParent = pe + (refRNG.randInt(p - pe - 1));
528 
529  // Mate:
530  for(j = 0; j < n; ++j) {
531  const unsigned sourceParent = ((refRNG.rand() < rhoe) ? eliteParent : noneliteParent);
532 
533  next(i, j) = curr(curr.fitness[sourceParent].second, j);
534 
535  //next(i, j) = (refRNG.rand() < rhoe) ? curr(curr.fitness[eliteParent].second, j) :
536  // curr(curr.fitness[noneliteParent].second, j);
537  }
538 
539  typedef Chromosome::ChromosomeType LocalChrType;
540  LocalChrType type = LocalChrType::OS_OR;
541 
542  if(curr.population[eliteParent].type == LocalChrType::RANDOM &&
543  curr.population[noneliteParent].type == LocalChrType::RANDOM)
544  type = LocalChrType::OS_RR;
545  else
546  if(curr.population[eliteParent].type != LocalChrType::RANDOM &&
547  curr.population[noneliteParent].type != LocalChrType::RANDOM)
548  type = LocalChrType::OS_OO;
549 
550  next.setType(i, type);
551  ++i;
552  }
553 
554  // We'll introduce 'pm' mutants:
555  while(i < p) {
556  //TODO: fix this.
557  for(j = 0; j < n; ++j) {
558  next(i, j) = Allele(refRNG.rand());
559  //next(i, j) = refRNG.randInt(n, unsigned(-(n + 1)));
560  //if(j < cut_point)
561  // next(i, j) = refRNG.randInt(left_lb, left_ub);
562  //else
563  // next(i, j) = refRNG.randInt(right_lb, right_ub);
564  }
565 
566  next.setType(i, Chromosome::ChromosomeType::RANDOM);
567  ++i;
568  }
569 
570  // Time to compute fitness, in parallel:
571  #ifdef _OPENMP
572  #pragma omp parallel for num_threads(MAX_THREADS) schedule(static,1)
573  #endif
574  for(int ii = int(pe); ii < int(p); ++ii) {
575  next.setFitness(ii, refDecoder.decode(next.population[ii]) );
576  }
577 
578  // Now we must sort 'current' by fitness, since things might have changed:
579  next.sortFitness(maximize);
580 }
