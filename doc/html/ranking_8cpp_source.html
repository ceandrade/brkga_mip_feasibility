105 void FractionalityRanker::setCurrentState(const std::vector<double>& x)
106 {
107  // calculate scores
108  for (unsigned int i = 0; i < integers.size(); i++)
109  {
110  double s = integralityViolation(x[integers[i]]); // number between 0 and 0.5
111  if (domain->varType(integers[i]) == 'B') s -= 10; // binaries score is always less than general integers
112  scores[i] = Score(s, integers[i]);
113  perm[i] = i;
114  }
115  // sort
116  if (reverse) permShellSort(&scores[0], &perm[0], scores.size(), std::greater<Score>(), true);
117  else permShellSort(&scores[0], &perm[0], scores.size(), std::less<Score>(), true);
118  // perturbe sorting
119  if (isNotNull(rankNoise) && (nCalled > noiseAfter))
120  {
121  int n = perm.size();
122  int nSwaps = int(rankNoise * n);
123  for (int i = 0; i < nSwaps; i++)
124  {
125  int fromIdx = rnd(n);
126  int toIdx = rnd(n);
127  std::swap(perm[fromIdx], perm[toIdx]);
128  }
129  }
130  nextItr = 0;
131  nCalled++;
132 }
